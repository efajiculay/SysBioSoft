from BioSANS2020 import biosans_lib as biosans
vcgn = """
    #REACTIONS
        X <=> 2 Y, 512, 512
    X + Y <=> Z  ,   8,   8
    Y + Z <=> X  ,   1,   1

    @CONCENTRATION
    X, 0.5
    Y, 0.5
    Z, 0.5
"""

my_model = biosans.model(vcgn)
data = my_model.run(method="Euler-2",tend=1,step_size_scaler=0.001,steps=1000,implicit=False)
my_model.clean()

from BioSANS2020 import biosans_lib as biosans

modelA = """
    Function_Definitions:
    alp = 0.15
    rx  = 0.12
    ry  = 0.7
    kx  = 0.5
    ky  = 0.1
    Yo  = 0.7
    Zo  = 0
    theta = lambda x,k,n : (x**n)/(x**n+k**n+sin(Zo))

    #REACTIONS
    0 NONE =>  X, 0
    0 NONE =>  timer_X, 1

    0 NONE =>  Y, 0  ::::: lambda X, Y    : -alp*Y + rx*theta(X,kx,2)
    0 NONE =>  Z, 0  ::::: lambda X, Y, Z : -alp*Z + rx*theta(X,kx,2)*ry*theta(Y,ky,2)


    @CONCENTRATION
    timer_X,   0, lambda timer_X  :  0 if timer_X >= 32 else timer_X
    X,         1, lambda timer_X  :  0 if timer_X >= 16 else 1
    Y,         Yo
    Z,         Zo
    NONE, 0 
"""
my_model = biosans.model(modelA)
data = my_model.run()

from BioSANS2020 import biosans_lib as biosans
modelA = """
    #REACTIONS, Volume = 1, tend = 100, steps = 100, FileUnit = molar

    -Sa    => mA        , 100
    mA     => 0 phi     , 1
    mA     => A + mA    , 1
    A      => 0 phi     , 1

    -Sb    => mB        , 100
    mB     => 0 phi     , 1
    mB     => B + mB    , 1
    B      => 0 phi     , 1

    -Sc    => mC        , 100
    mC     => 0 phi     , 1
    mC     => C + mC    , 1
    C      => 0 phi     , 1  

    @CONCENTRATION

    mA    , 1
    mB    , 0
    mC    , 0

    A     , 0
    B     , 0
    C     , 0

    phi   , 0
    -Sa   , 0   ,lambda C : 1/(1+C**2)
    -Sb   , 0   ,lambda A : 1/(1+A**2)
    -Sc   , 0   ,lambda B : 1/(1+B**2)
"""
my_model = biosans.model(modelA).save_traj("repressilator.txt").plot()
data = my_model.run()
data = my_model.run(tend=1000)
my_model.clean()
